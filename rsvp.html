        // --- SNAPSHOT HANDLERS ---

        function handleRsvpSnapshot(snapshot) {

            let totalGuests = 0;

            const guestHtml = [];

            const guestbookHtml = [];

            const playlistLinks = [];

            const sortedDocs = snapshot.docs.sort((a,b) => b.data().timestamp.toDate() - a.data().timestamp.toDate());

            sortedDocs.forEach(doc => {

                const rsvp = doc.data();

                const guestsInParty = (parseInt(rsvp.adults) || 0) + (parseInt(rsvp.kids) || 0);

                totalGuests += guestsInParty;

               

                guestHtml.push(`<span class="inline-flex items-center bg-blue-100 text-blue-800 text-sm font-medium px-2.5 py-1 rounded-full">${CONTRIBUTION_ICONS[rsvp.contributionType] || ''}<span class="ml-1.5">${rsvp.fullName} (${guestsInParty})</span></span>`);

               

                if (rsvp.message) {

                    guestbookHtml.push(`<div class="guestbook-card p-3 rounded-md"><p class="text-gray-700">"${rsvp.message}"</p><p class="text-right text-sm font-bold text-dark-blue/80 mt-1">- ${rsvp.fullName}</p></div>`);

                }

                if (rsvp.youtubeLinks && Array.isArray(rsvp.youtubeLinks)) {

                    playlistLinks.push(...rsvp.youtubeLinks);

                }

            });

           

            document.getElementById('total-guests').textContent = totalGuests;

            document.getElementById('guest-list').innerHTML = guestHtml.length ? guestHtml.join('') : '<p class="text-gray-500 text-sm">No RSVPs yet.</p>';

            document.getElementById('guestbook-section').innerHTML = guestbookHtml.length ? guestbookHtml.join('') : '<p class="text-gray-500 text-sm">No messages yet. Be the first!</p>';

            renderPlaylist(playlistLinks);

        }

        function handleItemsSnapshot(snapshot) {

            const broughtItems = {};

            claimedItemsCache = [];

            snapshot.docs.forEach(doc => {

                const item = doc.data();

                if (!broughtItems[item.category]) broughtItems[item.category] = [];

                broughtItems[item.category].push({ name: item.itemName, person: item.broughtBy });

                claimedItemsCache.push(item.itemName);

            });

            renderBroughtItems(broughtItems);

            renderNeededItems(broughtItems);

        }

        }

        function getYouTubeID(url) {

            const regExp = /^.*(youtu.be\/|v\/|u\/\w\/|embed\/|watch\?v=|&v=)([^#&?]*).*/;

            const match = url.match(regExp);

            return (match && match[2].length === 11) ? match[2] : null;

        }

        async function handleFormSubmit(e) {

            e.preventDefault();

            if (!isAuthReady) {

                showModal('Error', 'Database not ready. Please wait and try again.');

                return;

            }

           

            const submitBtn = document.getElementById('submit-btn');

            const formLoader = document.getElementById('form-loader');

            submitBtn.disabled = true;

            formLoader.classList.remove('hidden');

            const formData = new FormData(e.target);

            const fullName = formData.get('fullName').trim();

            const contributionType = formData.get('contributionType');

            const youtubeLinks = formData.get('youtubeLinks').split('\n').filter(link => link.trim() !== '');

            try {

                const batch = writeBatch(db);

                const rsvpRef = doc(collection(db, `artifacts/${appId}/public/data/rsvps`));

                batch.set(rsvpRef, {

                    fullName, contributionType,

                    adults: formData.get('adults'),

                    kids: formData.get('kids'),

                    email: formData.get('email'),

                    phone: formData.get('phone'),

                    message: formData.get('messageForTia'),

                    youtubeLinks: youtubeLinks,

                    timestamp: new Date(),

                    ownerId: userId

                });

                if (contributionType === 'money') {

                    const hallAmount = parseFloat(formData.get('hallDonation'));

                    if (hallAmount > 0) {

                        const hallRef = doc(collection(db, `artifacts/${appId}/public/data/hallDonations`));

                        batch.set(hallRef, { donorName: fullName, amount: hallAmount, rsvpId: rsvpRef.id });

                    }

                    const cookingAmount = parseFloat(formData.get('cookingDonation'));

                    if (cookingAmount > 0) {

                        const cookingRef = doc(collection(db, `artifacts/${appId}/public/data/cookingDonations`));

                        batch.set(cookingRef, { donorName: fullName, amount: cookingAmount, rsvpId: rsvpRef.id });

                    }

                } else {

                    const itemsRef = doc(collection(db, `artifacts/${appId}/public/data/potluckItems`));

                    let itemName, category;

                    if (contributionType === 'item') {

                        itemName = formData.get('itemToBring');

                        if(itemName) category = document.getElementById('itemToBring').querySelector(`option[value="${itemName}"]`).dataset.category;

                    } else if (contributionType === 'custom') {

                        itemName = formData.get('customItemName');

                        category = formData.get('customItemCategory');

                    }

                    if (itemName && category) {

                        batch.set(itemsRef, { itemName, category, broughtBy: fullName, rsvpId: rsvpRef.id });

                    }

                }

               

                await batch.commit();

                showModal('Thank You!', "Your RSVP is confirmed. We're so excited to celebrate with you!");

                e.target.reset();

                updateDonationSuggestion();

            } catch (error) {

                console.error("Error submitting RSVP:", error);

                showModal('Error', 'There was an error submitting your RSVP. Please try again.');

            } finally {

                submitBtn.disabled = false;

                formLoader.classList.add('hidden');

            }

        }

        async function generateMessage(e) {

            const btn = e.target;

            btn.disabled = true;

            btn.textContent = 'Thinking...';

            const prompt = "Generate three short, heartfelt and fun message ideas for a family member named Tia who is returning home for a celebration potluck. Return the response as a JSON array of strings.";

            try {

                const apiKey = "";

                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:generateContent?key=${apiKey}`;

                const payload = { contents: [{ parts: [{ text: prompt }] }] };

                const response = await fetch(apiUrl, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify(payload)

                });

                const result = await response.json();

                const textResponse = result.candidates[0].content.parts[0].text;

                const jsonString = textResponse.match(/\[.*\]/s)[0];

                const messages = JSON.parse(jsonString);

               

                const messageArea = document.getElementById('messageForTia');

                messageArea.value = messages[Math.floor(Math.random() * messages.length)];

            } catch (error) {

                console.error("Error generating message:", error);

                alert("Sorry, couldn't generate a message right now.");

            } finally {

                btn.disabled = false;

                btn.textContent = '✨ Get Message Ideas';

            }

        }

        async function findRsvp() {

            const nameInput = document.getElementById('find-name-input');

            const fullName = nameInput.value.trim();

            if (!fullName) {

                alert('Please enter a name.');

                return;

            }

            const rsvpsRef = collection(db, `artifacts/${appId}/public/data/rsvps`);

            const q = query(rsvpsRef, where("fullName", "==", fullName), where("ownerId", "==", userId));

           

            const querySnapshot = await getDocs(q);

            if (querySnapshot.empty) {

                showModal('Not Found', 'We could not find an RSVP with that name associated with your session. Please check the name or create a new RSVP.');

            } else {

                const rsvpDoc = querySnapshot.docs[0];

                showRsvpDetails(rsvpDoc);

            }

        }

        function showRsvpDetails(rsvpDoc) {

            const rsvpData = rsvpDoc.data();

            const content = `

                <h3 class="text-xl font-bold text-dark-blue mb-4">Your RSVP for ${rsvpData.fullName}</h3>

                <p><strong>Guests:</strong> ${rsvpData.adults} Adults, ${rsvpData.kids} Kids</p>

                <p class="text-gray-600 text-sm mt-4">If you need to make changes, please delete this RSVP and submit a new one.</p>

                <div class="mt-6 flex justify-between">

                    <button id="cancel-manage-btn" class="px-4 py-2 bg-gray-200 rounded-md hover:bg-gray-300">Cancel</button>

                    <button id="delete-rsvp-btn" data-id="${rsvpDoc.id}" class="px-4 py-2 bg-red-600 text-white rounded-md hover:bg-red-700">Delete RSVP</button>

                </div>

            `;

            showModal('Your RSVP Details', content, false);

            document.getElementById('delete-rsvp-btn').onclick = () => deleteRsvp(rsvpDoc.id);

            document.getElementById('cancel-manage-btn').onclick = closeModal;

        }

        async function deleteRsvp(rsvpId) {

            if (!confirm('Are you sure you want to delete this RSVP? This action cannot be undone.')) {

                return;

            }

            try {

                const batch = writeBatch(db);

                const rsvpDocRef = doc(db, `artifacts/${appId}/public/data/rsvps`, rsvpId);

                batch.delete(rsvpDocRef);

                const itemQuery = query(collection(db, `artifacts/${appId}/public/data/potluckItems`), where("rsvpId", "==", rsvpId));

                const itemSnapshot = await getDocs(itemQuery);

                itemSnapshot.forEach(doc => batch.delete(doc.ref));

                const cookingDonationQuery = query(collection(db, `artifacts/${appId}/public/data/cookingDonations`), where("rsvpId", "==", rsvpId));

                const cookingDonationSnapshot = await getDocs(cookingDonationQuery);

                cookingDonationSnapshot.forEach(doc => batch.delete(doc.ref));

               

                const hallDonationQuery = query(collection(db, `artifacts/${appId}/public/data/hallDonations`), where("rsvpId", "==", rsvpId));

                const hallDonationSnapshot = await getDocs(hallDonationQuery);

                hallDonationSnapshot.forEach(doc => batch.delete(doc.ref));

               

                await batch.commit();

                showModal('Success', 'Your RSVP has been successfully deleted.');

            } catch (error) {

                console.error("Error deleting RSVP: ", error);

                showModal('Error', 'Could not delete the RSVP. Please try again.');

            }

        }

       

        function updateDonationSuggestion() {

            const adultsInput = document.getElementById('adults');

            const kidsInput = document.getElementById('kids');

            const donationSuggestionLabel = document.getElementById('donation-suggestion-label');

            const cookingDonationInput = document.getElementById('cookingDonation');

            if(!adultsInput || !kidsInput || !donationSuggestionLabel || !cookingDonationInput) return;

            const adults = parseInt(adultsInput.value) || 0;

            const kids = parseInt(kidsInput.value) || 0;

            const suggestedAmount = (adults * 20) + (kids * 15);

            if (suggestedAmount > 0) {

                const amountString = suggestedAmount.toFixed(2);

                donationSuggestionLabel.textContent = `Trice's Cooking Fund (Suggested: $${amountString})`;

                cookingDonationInput.placeholder = amountString;

                cookingDonationInput.value = amountString;

            } else {

                donationSuggestionLabel.textContent = "Trice's Cooking Fund";

                cookingDonationInput.placeholder = '0.00';

                cookingDonationInput.value = '';

            }

        }

        function handleCountdown() {

            const countdownTimerEl = document.getElementById('countdown-timer');

            const countdownContainer = document.getElementById('countdown-container');

            const interval = setInterval(() => {

                const now = new Date();

               

                if (now > EVENT_DATE) {

                    clearInterval(interval);

                    countdownTimerEl.innerHTML = '<span class="text-green-600">It\'s Party Time!</span>';

                    countdownContainer.querySelector('p').textContent = 'Hope you\'re having fun!';

                    return;

                }

                if (now > RSVP_CLOSE_DATE) {

                     countdownContainer.querySelector('p').textContent = 'RSVPs are closed, but the party is almost here!';

                     document.getElementById('rsvp-form').innerHTML = '<p class="text-center font-bold text-red-600">RSVPs are now closed.</p>';

                     document.getElementById('manage-rsvp-btn').style.display = 'none';

                }

                const diff = EVENT_DATE - now;

                const days = Math.floor(diff / (1000 * 60 * 60 * 24));

                const hours = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));

                const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));

               

                countdownTimerEl.innerHTML = `

                    <span class="font-extrabold">${days}</span>d :

                    <span class="font-extrabold">${hours}</span>h :

                    <span class="font-extrabold">${minutes}</span>m

                `;

            }, 1000);

        }

        function showModal(title, content, showCloseButton = true) {

            modalContentEl.innerHTML = `

                <h3 class="text-xl font-bold text-dark-blue mb-4">${title}</h3>

                <div>${content}</div>

                ${showCloseButton ? `<div class="mt-6 text-right"><button id="modal-close-btn" class="px-4 py-2 bg-gray-200 rounded-md hover:bg-gray-300">Close</button></div>` : ''}

            `;

            modalEl.classList.remove('hidden');

            setTimeout(() => {

                modalEl.classList.remove('opacity-0');

                modalEl.querySelector('.modal-container').classList.remove('scale-95');

            }, 10);

            if(showCloseButton) {

                document.getElementById('modal-close-btn').onclick = closeModal;

            }

        }

        function closeModal() {

            modalEl.classList.add('opacity-0');

            modalEl.querySelector('.modal-container').classList.add('scale-95');

            setTimeout(() => modalEl.classList.add('hidden'), 300);

        }
